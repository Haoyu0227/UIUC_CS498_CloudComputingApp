# create heros table
CREATE TABLE heroes
(
    id              INT unsigned NOT NULL AUTO_INCREMENT,
    hero            VARCHAR(150) NOT NULL,
    power           VARCHAR(150) NOT NULL,
    name            VARCHAR(150) NOT NULL,
    xp              INT unsigned NOT NULL,
    color           VARCHAR(150) NOT NULL,
    PRIMARY KEY     (id)
);
